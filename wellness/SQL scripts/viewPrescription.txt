CREATE PROCEDURE `viewPrescription_procedure`(In user_Id int)
BEGIN
set @userId=(select userId from user where userId=user_Id);
select * from Prescription where prescriptionid IN (SELECT PrescriptionId FROM Prescription WHERE ProcedureId IN (SELECT ProcedureId FROM Procedures WHERE AppointmentId IN (SELECT AppointmentId from Appointment WHERE UserId=user_Id)));
END$$
